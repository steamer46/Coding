## 아이디어

각 정점에서 다른 모든 정점까지의 최소거리를 알아야 하므로, 플로이드 워셜이 가장 적합해 보였습니다, 이외에도 양의 가중치와 최소거리 구하기로 다익스트라 알고리즘도 가능할거 같아서 두가지 방법으로 풀어볼 수 있다고 생각했습니다





## 첫 번째 방법

이후에 코드에 살을 붙혀서 완성하면 다음과 같습니다

```cpp

#include <iostream>

#include <vector>

#include <algorithm>

using namespace std;

​

const int INF = 1e9;

int dist[101][101];

int item[101];

​

int main() {

    int n, m, r;

    cin >> n >> m >> r;

    for (int i = 1; i <= n; i++) {

        cin >> item[i];

        for (int j = 1; j <= n; j++)

            dist[i][j] = (i == j ? 0 : INF);

    }

​

    for (int i = 0; i < r; i++) {

        int a, b, l;

        cin >> a >> b >> l;

        dist[a][b] = l;

        dist[b][a] = l;

    }

​

    // Floyd-Warshall

    for (int k = 1; k <= n; k++)

        for (int i = 1; i <= n; i++)

            for (int j = 1; j <= n; j++)

                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);

​

    int result = 0;

    for (int i = 1; i <= n; i++) {

        int sum = 0;

        for (int j = 1; j <= n; j++) {

            if (dist[i][j] <= m)

                sum += item[j];

        }

        result = max(result, sum);

    }

    cout << result;

}
```


## 두 번째 방법

다익스트라를 이용하는 방법입니다.
각각 정점에서 최단거리를 구해야 하는것이고, 양의 가중치를 가지고 있기 때문에, 다익스트라 알고리즘을 이용할 수 있습니다

​

기본적인 다익스트라 알고리즘에다가 dist가 m보다 작으면 모두 sum에다가 더해준다는 코드를 넣어서 위에 플로이드 워셜과 같은 방식으로 코드를 작성해 줍니다.

​

기본적인 다익스트라 알고리즘 이기 때문에 별 설명 없이 코드만 적겠습니다


```cpp

#include<iostream>

#include<vector>

#include<queue>

using namespace std;

​

const int INF =1e9;

int item[101];

int dist[101];

int n,m,r;

int sum= 0;

vector<pair<int, int >> graph[101];

​

void dijkstra(int start){

for(int i= 1; i<= n ; i++) dist[i] = INF;

priority_queue< pair<int,int>, vector<pair<int,int>>, greater<>>pq;

​

dist[start]=0;

pq.push({0,start});

​

while(!pq.empty()){

auto[cost, now]= pq.top();

pq.pop();

​

if(dist[now] < cost) continue;

​

for(auto[next, weight]: graph[now]){

int nextDist =  cost+weight;

if(nextDist <dist[next]){

dist[next]= nextDist;

pq.push({nextDist, next });

}

}

}

​

sum =0;

for(int i =1; i<=n; i++){

if (dist[i]<= m) sum+= item[i];

}

}

​

int main() {

int result = 0;

cin>> n>>m>>r;

​

for(int i=1; i<=n; i++){

cin>>item[i];

}

​

for (int i=0; i<r; i++){

int a, b, c;

cin >> a >> b >>c;

graph[a].push_back({b,c});

graph[b].push_back({a,c});

}

​

for(int i=1; i<=n; i++){

dijkstra(i);

result = max(sum, result);

}

​

cout<< result;

return 0;

}
```

다익스트라 알고리즘, 플로이드 워셜을 이용한 알고리즘 둘다 정답이 나오게 됩니다!
