```cpp
#include <iostream>
#include <queue>
#include <tuple>
#include <cstring>

using namespace std;

int N, M, T; 
int map[101][101]; 
int dx[4] = {-1, 1, 0, 0}; 
int dy[4] = {0, 0, -1, 1}; 
int visited[101][101][2];

int main() {
    ios::sync_with_stdio(false); 
    cin.tie(0); 
    memset(visited, -1, sizeof(visited)); 
    cin >> N >> M >> T;

    for (int i = 1; i <= N; i++) {
        for (int j = 1; j <= M; j++) {
            cin >> map[i][j];
        }
    }

    queue<tuple<int, int, int>> q;
    visited[101][101][0] = 0;
    q.push({1, 1, 0});
    int ans = 1e9;
    while (!q.empty()) {
        auto [y, x, sword] = q.front(); q.pop();
        int time = visited[y][x][sword];
        if (time > T) continue;
        if (y == N && x == M) {
            ans = time;
            break;
        }

        for (int d = 0; d < 4; d++) {
            int ny = y + dy[d];
            int nx = x + dx[d];
            if (ny < 1 || ny > N || nx < 1 || nx > M) continue;
            
            // 0: 빈칸
            if (map[ny][nx] == 0 && visited[ny][nx][sword] == -1) {
                visited[ny][nx][sword] = time + 1;
                q.push({ny, nx, sword});
            }

            // 2: 검
            if (map[ny][nx] == 2 && visited[ny][nx][1] == -1) {
                visited[ny][nx][1] = time + 1;
                q.push({ny, nx, 1}); // 검 획득
            }

            // 1: 벽 - 검이 있을 때만 통과 가능
            if (map[ny][nx] == 1 && sword == 1 && visited[ny][nx][1] == -1) {
                visited[ny][nx][1] = time + 1;
                q.push({ny, nx, 1});
            }
        }
    }

    if (ans <= T) cout << ans << '\n';
    else cout << "Fail" << '\n';

    return 0;
}

```
