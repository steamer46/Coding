위와 같은 방식을 이용해서 분할정복을 이용해주면 됩니다.
위 같은 예시를 중심으로 아래 코드를 완성시키면 다음과 같습니다

```cpp
​

#include <iostream>

using namespace std;

const int MAX = 128;

int graph[MAX][MAX];

int white = 0, blue = 0;

​

void z(int r, int c, int n) {

    bool allSame = true;

    for (int i = r; i < r + n; i++) {

        for (int j = c; j < c + n; j++) {

            if (graph[i][j] != graph[r][c]) {

                allSame = false;

                break;

            }

        }

        if (!allSame) break;

    }

​

    if (allSame) {

        if (graph[r][c] == 0) white++;

        else blue++;

        return;

    }

​

    int k = n / 2;

    z(r, c, k);          // 왼쪽 위

    z(r, c + k, k);      // 오른쪽 위

    z(r + k, c, k);      // 왼쪽 아래

    z(r + k, c + k, k);  // 오른쪽 아래

​

}

​

int main() {

    int N;

    cin >> N;

    for (int i = 0; i < N; i++) {

        for (int j = 0; j < N; j++) {

            cin >> graph[i][j];

        }

    }

    z(0, 0, N);

    cout << white << "\n" << blue << "\n";

    return 0;

}
```
